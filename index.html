// ===== SIMPLE IN-MEMORY RATE LIMIT ===== const rateLimitMap = new Map(); function isRateLimited(ip) { const now = Date.now(); const windowMs = 10 * 60 * 1000; // 10 minutes const maxRequests = 10; if (!rateLimitMap.has(ip)) { rateLimitMap.set(ip, []); } const timestamps = rateLimitMap.get(ip).filter(ts => now - ts < windowMs); timestamps.push(now); rateLimitMap.set(ip, timestamps); return timestamps.length > maxRequests; } export default async function handler(req, res) { if (req.method !== "POST") { return res.status(405).json({ error: "Method not allowed" }); } // ===== SAFE IP DETECTION ===== const ip = req.headers["x-forwarded-for"]?.split(",")[0]?.trim() || req.socket?.remoteAddress || "unknown"; // ===== RATE LIMIT CHECK ===== if (isRateLimited(ip)) { return res.status(429).json({ error: "Too many requests. Please wait 10 minutes." }); } const { prompt } = req.body; // ===== VALIDATION ===== if (!prompt || typeof prompt !== "string") { return res.status(400).json({ error: "Invalid prompt" }); } if (prompt.length > 500) { return res.status(400).json({ error: "Prompt too long (max 500 chars)" }); } const FAL_KEY = process.env.FAL_KEY; // ===== DEMO IMAGE (Always Safe Fallback) ===== const demoImage = "https://picsum.photos/1024"; // ===== NO API KEY → DEMO MODE ===== if (!FAL_KEY) { return res.status(200).json({ url: demoImage }); } try { const response = await fetch("https://fal.run/fal-ai/fast-sdxl", { method: "POST", headers: { "Authorization": `Key ${FAL_KEY}`, "Content-Type": "application/json" }, body: JSON.stringify({ prompt: prompt, image_size: "square_hd" }) }); // ===== API STATUS CHECK ===== if (!response.ok) { console.error("FAL API error status:", response.status); return res.status(200).json({ url: demoImage }); } const data = await response.json(); // ===== SUCCESS HANDLING ===== if (data?.images?.[0]?.url) { return res.status(200).json({ url: data.images[0].url }); } if (data?.output?.[0]?.url) { return res.status(200).json({ url: data.output[0].url }); } if (typeof data?.url === "string") { return res.status(200).json({ url: data.url }); } console.error("Unexpected API format:", data); return res.status(200).json({ url: demoImage }); } catch (error) { console.error("Server crash:", error); return res.status(200).json({ url: demoImage }); } }
